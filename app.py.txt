import pandas as pd
import numpy as np
import plotly.express as px
from dash import Dash, dcc, html
import os

# Chargement - Adapter le chemin pour Render
df = pd.read_csv("C:/Users/HP/Desktop/Business Analysis/SuperStore Sales/Superstore.csv", 
                 encoding='latin-1')

# Nettoyage
df['Order Date'] = pd.to_datetime(df['Order Date'], format='%d-%m-%Y', errors='coerce')
df['Ship Date'] = pd.to_datetime(df['Ship Date'], format='%d-%m-%Y', errors='coerce')
df = df.dropna(subset=['Order Date'])

# Creation des colonnes
df['Profit_Margin'] = (df['Profit'] / df['Sales']) * 100
df['Year'] = df['Order Date'].dt.year
df['Month'] = df['Order Date'].dt.month
df['Month_Name'] = df['Order Date'].dt.strftime('%b')
df['Discount_Category'] = pd.cut(df['Discount'], 
                                  bins=[-0.001, 0.10, 0.20, 0.30, 1.0],
                                  labels=['0-10% Safe', '10-20% Optimal', 
                                          '20-30% Dangerous', '>30% Critical'])

# Calculs des indicateurs
nb_annees = df['Year'].nunique()
total_sales = df['Sales'].sum()
total_profit = df['Profit'].sum()
avg_margin = df['Profit_Margin'].mean()
loss_transactions = len(df[df['Profit'] < 0])
loss_percentage = (loss_transactions / len(df)) * 100
critical_discount_loss = df[df['Discount'] > 0.20]['Profit'].sum()

# AgrÃ©gations
profit_by_category = df.groupby('Category')['Profit'].sum().reset_index()
profit_by_segment = df.groupby('Segment')['Profit'].sum().reset_index()
top_losing = df[df['Profit'] < 0].groupby('Sub-Category')['Profit'].sum().sort_values().head(5).reset_index()

# Analyse des remises
discount_analysis = df.groupby('Discount_Category', observed=False).agg({
    'Profit_Margin': 'mean',
    'Sales': 'sum',
    'Profit': 'sum'
}).reset_index()
discount_analysis.columns = ['Discount_Category', 'Avg_Margin_Pct', 'Total_Sales', 'Total_Profit']

# Serie temporelle
time_series = df.groupby(['Year', 'Month'])['Profit'].sum().reset_index()
time_series['Date'] = pd.to_datetime(time_series['Year'].astype(str) + '-' + 
                                      time_series['Month'].astype(str) + '-01')

# Calculs des gains (donnees reelles)
remise_excessive = df[df['Discount'] > 0.20]
ca_excess_annuel = remise_excessive['Sales'].sum() / nb_annees
perte_excess_annuelle = remise_excessive['Profit'].sum() / nb_annees
taux_conservation = 0.50
gain_option_c = (abs(perte_excess_annuelle) * taux_conservation) + (ca_excess_annuel * taux_conservation * 0.175)

tables_data = df[df['Sub-Category'] == 'Tables']
bookcases_data = df[df['Sub-Category'] == 'Bookcases']
ca_tables_annuel = tables_data['Sales'].sum() / nb_annees
ca_bookcases_annuel = bookcases_data['Sales'].sum() / nb_annees
taux_reduction = 0.10
gain_option_b = (ca_tables_annuel * taux_reduction) + (ca_bookcases_annuel * taux_reduction)

segment_analysis = df.groupby('Segment')['Profit_Margin'].mean()
consumer_margin = segment_analysis.get('Consumer', 0)
corporate_margin = segment_analysis.get('Corporate', 0)
ca_consumer_annuel = df[df['Segment'] == 'Consumer']['Sales'].sum() / nb_annees
margin_gap = corporate_margin - consumer_margin
gain_option_a = ca_consumer_annuel * (margin_gap / 100) * 0.20

total_gain = gain_option_c + gain_option_b + gain_option_a
